import pandas as pd import os # 1. Carica e unisce i 4 file delle quotazioni ufficiali (saltando la prima riga descrittiva) file_ruoli = [ 'Quotazioni_Fantacalcio_Stagione_2025_26.xlsx - Portieri.csv', 'Quotazioni_Fantacalcio_Stagione_2025_26.xlsx - Difensori.csv', 'Quotazioni_Fantacalcio_Stagione_2025_26.xlsx - Centrocampisti.csv', 'Quotazioni_Fantacalcio_Stagione_2025_26.xlsx - Attaccanti.csv' ] lista_quotazioni = [] for file in file_ruoli: if os.path.exists(file): # Carica il listone saltando la riga iniziale di intestazione generica df_ruolo = pd.read_csv(file, skiprows=1) # Teniamo solo le colonne fondamentali, pulendo i nomi da spazi bianchi df_ruolo.columns = df_ruolo.columns.str.strip() df_ruolo = df_ruolo[['R', 'Nome', 'Squadra', 'Qt.A']].copy() lista_quotazioni.append(df_ruolo) else: print(f"Errore: Il file '{file}' non è presente nella cartella.") # Unisce tutti i calciatori del listone ufficiale in un unico grande dizionario per una mappatura rapida listone_completo = pd.concat(lista_quotazioni, ignore_index=True) listone_completo['Nome_Clean'] = listone_completo['Nome'].str.lower().str.strip() listone_completo['Squadra_Clean'] = listone_completo['Squadra'].str.lower().str.strip() # Crea una mappa chiave -> quotazione usando (Nome, Squadra) per evitare qualsiasi errore di omonimia mappa_quotazioni = {} for _, row in listone_completo.iterrows(): chiave = (row['Nome_Clean'], row['Squadra_Clean'][:3]) # Prime 3 lettere della squadra per evitare disallineamenti (es. Inter/Int) mappa_quotazioni[chiave] = { 'Ruolo_Reale': row['R'], 'Nome_Reale': row['Nome'], 'Squadra_Reale': row['Squadra'], 'Quotazione': pd.to_numeric(row['Qt.A'], errors='coerce') } # 2. Carica il file reale delle rose della tua lega file_rose_lega = 'Rose_muyfantacalcio-league.xlsx - TutteLeRose.csv' if not os.path.exists(file_rose_lega): print(f"Errore fatale: Il file delle rose '{file_rose_lega}' non è stato trovato!") exit() # Legge il file riga per riga per estrarre la struttura corretta a due colonne affiancate rose_finali = [] fantasquadra_corrente_sinistra = "" fantasquadra_corrente_destra = "" with open(file_rose_lega, 'r', encoding='utf-8') as f: for linea in f: colonne = [c.strip() for c in linea.split(',')] if len(colonne) < 4: continue # Rileva i nomi delle fantasquadre nelle righe di intestazione dei blocchi if "Ruolo" not in colonne[0] and colonne[0] != "" and (len(colonne) > 1 and colonne[1] == ""): fantasquadra_corrente_sinistra = colonne[0] if len(colonne) >= 6 and colonne[5] != "": fantasquadra_corrente_destra = colonne[5] continue if "Ruolo" in colonne[0] or colonne[0] == "" or "Rose lega" in colonne[0] or "https" in colonne[0]: continue # Elabora la parte sinistra della riga (Fantasquadra 1) ruolo_sist = colonne[0] nome_sist = colonne[1] squadra_sist = colonne[2] if nome_sist and fantasquadra_corrente_sinistra: rose_finali.append({ 'Fantasquadra': fantasquadra_corrente_sinistra, 'Ruolo_Dichiarato': ruolo_sist, 'Calciatore': nome_sist, 'Squadra_SerieA': squadra_sist }) # Elabora la parte destra della riga (Fantasquadra 2, se presente) if len(colonne) >= 8: ruolo_d = colonne[5] nome_d = colonne[6] squadra_d = colonne[7] if nome_d and fantasquadra_corrente_destra: rose_finali.append({ 'Fantasquadra': fantasquadra_corrente_destra, 'Ruolo_Dichiarato': ruolo_d, 'Calciatore': nome_d, 'Squadra_SerieA': squadra_d }) df_rose_lega = pd.DataFrame(rose_finali) # 3. Incrocia i dati reali delle rose con il dizionario delle quotazioni dati_accorpati = [] for _, giocatore in df_rose_lega.iterrows(): nome_c = giocatore['Calciatore'].lower().strip() squadra_c = giocatore['Squadra_SerieA'].lower().strip()[:3] # Cerca la corrispondenza esatta nel listone ufficiale delle quotazioni match = mappa_quotazioni.get((nome_c, squadra_c)) if match: dati_accorpati.append({ 'Fantasquadra': giocatore['Fantasquadra'], 'Ruolo': match['Ruolo_Reale'], 'Calciatore': match['Nome_Reale'], 'Squadra_SerieA': match['Squadra_Reale'], 'Quotazione_Attuale': match['Quotazione'] }) else: # Se non trova la squadra abbreviata, prova a cercare solo per nome (gestione eccezioni) trovato_solo_nome = False for (n_m, s_m), v_m in mappa_quotazioni.items(): if n_m == nome_c: dati_accorpati.append({ 'Fantasquadra': giocatore['Fantasquadra'], 'Ruolo': v_m['Ruolo_Reale'], 'Calciatore': v_m['Nome_Reale'], 'Squadra_SerieA': v_m['Squadra_Reale'], 'Quotazione_Attuale': v_m['Quotazione'] }) trovato_solo_nome = True break if not trovato_solo_nome: # Salvataggio di sicurezza per i calciatori non trovati (es. fuori listone o ortografia diversa) dati_accorpati.append({ 'Fantasquadra': giocatore['Fantasquadra'], 'Ruolo': giocatore['Ruolo_Dichiarato'], 'Calciatore': giocatore['Calciatore'], 'Squadra_SerieA': giocatore['Squadra_SerieA'], 'Quotazione_Attuale': 0 }) # 4. Crea il DataFrame finale e lo ordina per Fantasquadra e Quotazione Decrescente df_output = pd.DataFrame(dati_accorpati) df_output.sort_values(by=['Fantasquadra', 'Quotazione_Attuale'], ascending=[True, False], inplace=True) # 5. Esporta il database unico riordinabile nome_file_output = 'Database_Quotazioni_Rose_Completo.csv' df_output.to_csv(nome_file_output, index=False, encoding='utf-8-sig') print(f"\nIncrocio completato correttamente dal file delle rose reale!") print(f"Il file '{nome_file_output}' è stato creato ed è pronto per essere aperto con LibreOffice Calc.")
